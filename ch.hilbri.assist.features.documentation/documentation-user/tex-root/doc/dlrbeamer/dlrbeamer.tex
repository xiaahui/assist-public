% DLR Theme Usage
\section{DLR Theme Usage}
\begin{frame}{DLR Theme Usage}
	\begin{itemize}
		\item Load the beamer class and the DLR theme: \\
		      \texttt{\textbackslash documentclass\{beamer\}}\\				
		      \texttt{\textbackslash usetheme\{dlr\}}	
		\item This will load the current DLR slide layout
		\item Note:					
						\begin{itemize}
							\item Use \texttt{\textbackslash usetheme[professionalfonts]\{dlr\}} to use the default LaTeX math fonts.
							\item Use \texttt{\textbackslash usetheme[helvetica]\{dlr\}} to use Helvetica instead of Frutiger.
							\item To use the old template use \texttt{\textbackslash usetheme\{dlr\_2008\}}.
						\end{itemize}
		\item For this document, an additional color theme is used: \texttt{\textbackslash usecolortheme\{orchid\}}.
		\alert{\item The theme is subject to change, as e.g. fonts will be adapted to the PPT template etc..!}
		\item For more information: See the beamer documentation at\\
		      \url{http://www.tex.ac.uk/CTAN/macros/latex/contrib/beamer/doc/beameruserguide.pdf}
		\item The package beamertools is loaded with DLR beamer:		
					\begin{itemize}
						\item To avoid the loading of beamertools use \texttt{\textbackslash usetheme[nobeamertools]\{dlr\}}
						\item To access beamertools options use \texttt{\textbackslash usetheme[beamertools=\{\dots\}]\{dlr\}}
					\end{itemize}		
	\end{itemize}
\end{frame}


% Additional Logo
\begin{frame}{Additional Logo}
	\begin{itemize}
		\item You can add an additional logo next to the DLR Eagle
		\item Use \texttt{\textbackslash addlogoFL\{xxx\}} to add some content right of the DLR Eagle in the footline
		\item Use \texttt{\textbackslash addlogoTP\{xxx\}} to add some content right of the DLR Eagle on the title page
		\item You can use  \texttt{\textbackslash addlogoFL*\{xxx\}} and \texttt{\textbackslash addlogoTP*\{xxx\}} to directly add some TikZ code. The node (DLReagleTP) and (DLReagleTP) refers to the DLR Eagle in the footline and title page, respectively.
	\end{itemize}
\end{frame}
