% Usage
\section{Usage}
\begin{frame}{Theme usage}
	\begin{itemize}
		\item Load the beamer class and the DLR theme: \\
		      \texttt{\textbackslash documentclass\{beamer\}}\\				
		      \texttt{\textbackslash usetheme\{dlr\_2008\}}
		\item This will load the generic DLR background image and base font.
		\item You can change the image used with \texttt{\textbackslash settitle\{area=...\}}. The option takes the values
		      \texttt{all,aeronautics,space,transport,energy}.
		\item For this document, an additional color theme is used: \texttt{\textbackslash usecolortheme\{orchid\}}.
		\alert{\item This template is obsolete! A new version is released!}
		\item For more information: See the beamer documentation at\\
		      \url{http://www.tex.ac.uk/CTAN/macros/latex/contrib/beamer/doc/beameruserguide.pdf}
	\end{itemize}
\end{frame}
